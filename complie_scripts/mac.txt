# Build command for macOS (clang + OpenMP)
# Author: Xuan Tung VU

PYTHON=python
EXT=$($PYTHON -c "import sysconfig; print(sysconfig.get_config_var('EXT_SUFFIX') or '.so')")

 
CONDA_PREFIX=$(python -c "import os; print(os.environ['CONDA_PREFIX'])")

rm -f viability*.so viability*.dylib 2>/dev/null || true

clang++ -O3 -std=c++17 -shared -fPIC -undefined dynamic_lookup \
  $($PYTHON -m pybind11 --includes) \
  -I. \
  -I"$CONDA_PREFIX/include" \
  ViabilityKernelCPU.cpp CaptureBasinCPU.cpp ROptionSetsComputer.cpp \
  -Xpreprocessor -fopenmp \
  -L"$CONDA_PREFIX/lib" -lomp \
  -Wl,-rpath,"$CONDA_PREFIX/lib" \
  -o viability$EXT
