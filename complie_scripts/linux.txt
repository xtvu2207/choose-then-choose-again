# Build command for Linux (g++ + OpenMP)
# Author: Xuan Tung VU

rm -f viability*.so 2>/dev/null

PYTHON=python3
EXT=$($PYTHON -c "import sysconfig; print(sysconfig.get_config_var('EXT_SUFFIX') or '.so')")

g++ -O3 -std=c++17 -shared -fPIC \
  $($PYTHON -m pybind11 --includes) \
  -I. \
  ViabilityKernelCPU.cpp \
  CaptureBasinCPU.cpp \
  ROptionSetsComputer.cpp \
  -fopenmp \
  -Wno-attributes \
  -o viability$EXT
